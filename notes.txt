2. Jupyter / notebook (one clip at a time)
import json
from pathlib import Path
from IPython.display import Image, display, Markdown
# Add pipeline root if caption_to_str import fails:
# import sys; sys.path.insert(0, "/mnt/data0/harsha/new_dataset_pipeline/Indic_video_pipeline")
from common.caption_text import caption_to_str
ws = Path("/mnt/data0/harsha/new_dataset_pipeline/pipeline_outputs/workspaces/devdas_standard")
clip_id = "devdas_standard_000001"
rec = next(json.loads(l) for l in ws.joinpath("metadata.jsonl").open() if clip_id in l)
img = ws / "actor_frames" / f"{clip_id}.jpg"
display(Image(filename=str(img), width=900))
actors = ", ".join(a["display_name"] for a in rec.get("actors", []))
display(Markdown(f"**{clip_id}** | {rec['timestamp_start']:.1f}s–{rec['timestamp_end']:.1f}s | {actors}\n\n{caption_to_str(rec['caption'])}")
3. Terminal-only (text check, no image)
python3 <<'PY'
import json
from pathlib import Path
sys_path = "/mnt/data0/harsha/new_dataset_pipeline/Indic_video_pipeline"
import sys; sys.path.insert(0, sys_path)
from common.caption_text import caption_to_str
ws = Path("/mnt/data0/harsha/new_dataset_pipeline/pipeline_outputs/workspaces/devdas_standard")
for line in ws.joinpath("metadata.jsonl").open():
    r = json.loads(line)
    if r.get("verdict") != "FINAL" or not r.get("caption"):
        continue
    img = ws / "actor_frames" / f"{r['clip_id']}.jpg"
    actors = [a.get("display_name") for a in r.get("actors", [])]
    print("=" * 60)
    print(r["clip_id"], img.exists(), actors)
    print(caption_to_str(r["caption"])[:500])
    break  # remove break to dump all
PY
Mapping: metadata.jsonl row → actor_frames/<clip_id>.jpg is the paired image for caption review. Captions can be a list of paragraphs (e.g. devdas_standard_000000); the viewer joins them into one string for reading.

Each full pipeline run now gets its own timestamped output folder (enabled in pipeline_v2.yaml).

New layout
v2_outputs/
  runs/
    devdas_standard_20260605_120000/
      run.json              # run metadata
      workspace/            # all clip outputs (metadata, export, actor_frames, …)
      logs/s1 … s12/
      reports/              # runtime JSON, report, runtime_summary.csv
    ABCD_20260605_143000/
      workspace/
      logs/
      reports/
Old flat v2_outputs/workspaces/devdas_standard/ is not overwritten on new runs.
